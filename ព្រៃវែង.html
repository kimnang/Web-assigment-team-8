ដំណើរកម្សាន្តនៅក្នុងខេត្តព្រៃវែង

ខេត្តព្រៃវែង ជាខេត្តដែលមានទីតាំងស្ថិតនៅភាគខាងកើតរាជធានីភ្នំពេញ មានព្រំប្រទល់ខាងកើតជាប់ខេត្តស្វាយរៀង និងវៀតណាម ខាងជើងជាប់ខេត្តកំពង់ចាម និងខេត្តត្បូងឃ្មុំ ខាងលិចជាប់ខេត្តកណ្តាល និងខាងត្បូងជាប់វៀតណាម។ ខេត្តព្រៃវែង មានច្រកអន្តរជាតិចំនួន ៣ កន្លែងគឺ ទី១ ច្រកអន្តរជាតិម៉ឺនជ័យ (ស្រុកកំចាយមារ) ទី២ ច្រកអន្តរជាតិបន្ទាយចក្រី (ស្រុកព្រះស្តេច) ទី៣ ច្រកអន្តរជាតិកោះរការ (ស្រុកពាមជរ) នឹងមានផ្ទៃដីសរុបគឺ ប្រហែល ៤៨៨៣ គម² ដែលស្មើទៅនិង ២,៧០% នៃផ្ទៃដីសរុបនៃប្រទេសកម្ពុជា (១៨១ ០៣៥ គម²), ដែលក្នុងនោះ ៤៤៥,១៨ គម² ឬ ៩,១២% ជាផ្ទៃដី ដែលប្រជាជន បានតាំងទីលំនៅ, ៣១០០ គម² ឬ ៦៣,៤៩% ជាដី សំរាប់កសិកម្ម, ១៩៤,៦១ គម² ឬ ៣,៩៩% ជាផ្ទៃនៃតំបន់ ព្រៃឈើ, ១០៨២,៨៦ គម² ឬ ២២,១៨% ជាដី សាធារណៈ, ហេដ្ឋារចនាសម្ព័ន្ធ និង ធារាសាស្រ្ត, និង នៅសល់ ៦០,៣៥ គម² ឬ១,២៤%ជាដីមិនត្រូវបានប្រើប្រាស់។

ទីកន្លែងកំសាន្ដ

ចុងប៉ម

ក្រុងព្រៃវែង ត្បិតខ្សត់កន្លែងកម្សាន្តមែន តែមានកន្លែងមួយដែលមើលថ្ងៃលិចស្អាតបំផុត បើបានឃើញហើយដឹងតែជាប់ចិត្តស្ទើរដកមិនរួច។ នោះគ្មានអីក្រៅពី ចុងប៉ម ល្បីប្រចាំក្រុងព្រៃវែងនោះឡើយ។អមដោយទេសភាពថ្ងៃលិចនារដូវទឹកឡើងដែលមិនចាញ់ភ្នំក្រោម និងភ្នំបាខែង ខេត្តសៀមរាប នោះ អ្នកក្រុងព្រៃវែង ទាំងសិស្ស ទាំងអ្នកធម្មតា តែងមកទីនេះស្រូបយកខ្យល់បរិសុទ្ធនៃក្លិនទឹកទន្លេ ត្រជាក់ស្វាងអស់ខួរក្បាល។ ឯខែប្រាំងវិញ គេនៅតែស្រូបខ្យល់បរិសុទ្ធត្រជាក់ តែអមដោយទេសភាពវាលស្រែប្រាំងវិញម្ដង ជំនួសលើផ្ទៃទឹកនាខែវស្សា។ ​


ភ្នំឈើកាច់​

រមណីយដ្ឋាននគរភ្នំ ឬរមណីយដ្ឋានភ្នំឈើកាច់ ជារមណីយដ្ឋានបែបប្រវត្ដិសាស្រ្ដដែលស្ថិតនៅលើភ្នំឈើកាច់ដោយមានកាំជណ្តើរឡើងដល់កំពូលភ្នំ។ រមណីយដ្ឋាននេះ ជាកន្លែងស័ក្កសិទ្ធិមួយក្នុងខេត្ដព្រៃវែង ដែលមានអ្នកមកគោរពបូជាជាច្រើន និងជាកន្លែងសម្រាប់ប្រារព្ធពិធីសាសនាផ្សេងៗ។ ជារៀងរាល់ឆ្នាំរមណីយដ្ឋាននេះតែងតែប្រារព្ធពិធីឡើងអ្នកតា (អ្នកតាមេសរ) ដែលប្រព្រឹត្ដរយៈពេលបីថ្ងៃ ដើម្បីរំលឹកគុណដល់បុព្វបុរស សុំសេចក្តីសុខ សុំទឹកភ្លៀងនិងការពារប្រជាពលរដ្ឋទាំងអស់នៅក្នុងភូមិឱ្យមានសេចក្តីសុខ។ ភ្ញៀវទេសចរអាចទៅកម្សាន្ដ ទស្សនាទេសភាពពីលើកំពូលភ្នំ ទស្សនាវាលស្រែខៀវស្រងាត់ដាច់កន្ទុយភ្នែកផងដែរ។​


បឹងស្នេហ៍

តំបន់អភិរក្សបឹងស្នេហ៍ គឺជាតំបន់អភិរក្សធម្មជាតិទំហំជាង៧០ហិកតា ស្ថិតនៅលើទីទួលពាន់កាឡី ក្នុងភូមិបឹងស្នេហ៍ ឃុំធាយ ស្រុកបាភ្នំ ខេត្តព្រៃវែង ។ តំបន់អភិរក្សនោះក្រៅពីមានផ្ទៃបឹងដ៏ធំ ក៏មានទីទួល ព្រៃរបោះ និងព្រៃលិចទឹក ជាពិសេសពពួកសត្វស្លាបច្រើនប្រភេទចំនួនរាប់ពាន់ក្បាលរស់នៅ។ ​


ប្រាសាទចុងស្រុក

ប្រាសាទចុងស្រុកខេត្តព្រៃវែងបង្កប់រឿងរ៉ាវចម្លែកច្រើន តែមិនសូវមានអ្នកស្គាល់ប្រាសាទចុងស្រុក! នេះ(អ្នកស្រុកហៅថាព្រៃដើមស្រុក ឬប្រធាតុ) ស្ថិតនៅភូមិបឹងជ័រ ឃុំព្រៃទឹង ស្រុកស៊ីធរកណ្ដាល ខេត្តព្រៃវែង។ ប្រាសាទដ៏ចំណាស់មួយនេះ ធ្លាប់មានរឿងរ៉ាវចម្លែកជាច្រើនកើតឡើង ប៉ុន្តែមិនសូវមានអ្នកបានដឹង បានស្គាល់ឡើយ ហើយបច្ចុប្បន្នកំពុងស្ថិតក្នុងសភាពស្ងាត់ជ្រងំសម្រាប់ថ្ងៃធម្មតា និងមានមនុស្សមួយចំនួនតូចបានទៅលេងនៅថ្ងៃសៅរ៍ ថ្ងៃអាទិត្យ។ ​
