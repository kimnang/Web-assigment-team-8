ដំណើរកម្សាន្តនៅក្នុងខេត្តស្វាយរៀង

ខេត្តស្វាយរៀងប្រហែលជាខេត្តមួយក្នុងចំណោមខេត្តតូចបំផុតរបស់ប្រទេសកម្ពុជា ប៉ុន្តែមានផ្លូវជាតិដែលពេញទៅដោយចរាចរណ៍មមាញឹកបំផុតកាត់តាមទីប្រជុំជន។ ផ្លូវជាតិលេខ១ភ្ជាប់ពីភ្នំពេញទៅទីក្រុងហូជីមិញរបស់ប្រទេសវៀតណាមបន្ទាប់ពីឆ្លងទន្លេមេគង្គតាមរយៈស្រឡាងអ្នកលឿង។ ទីនេះជាច្រកចូលអន្តរជាតិតាមច្រកបាវិតទៅកាន់ប្រទេសវៀតណាម។ នេះជាក្រុងមួយដ៏រីកចម្រើនដែលបានទទួលផលពីចរាចរណ៍ពាណិជ្ជកម្មតាមព្រំដែន។ ទីនោះមានទីផ្សារដ៏មមាញឹក ហើយប្រជាជនមានភាពរស់រាយរាក់ទាក់។ ខេត្តនេះមានជម្រើសសម្រាប់ស្នាក់អាស្រ័យពិសេសៗជាច្រើន។ នៅជិតទន្លេវ៉ៃកូ ក៏មានវាលភក់ដ៏ធំនិងមានទេសភាពស្អាតផ្ដល់ជាបរិស្ថានយ៉ាងល្អសម្រាប់ការថតរូប។ នៅមិនឆ្ងាយប៉ុន្មានពីទីក្រុង មានស្ពានមួយឆ្លងទន្លេវ៉ៃកូដែលផ្ដល់ជូនជាចំណងដៃពី សម្ដេចនាយករដ្ឋមន្រ្តី ហ៊ុន សែន ។ ភាពស្រស់ស្អាតនៃខេត្តនេះមិនមែនដោយសារតែភាពងាយស្រួលរបស់ខេត្តនេះប៉ុណ្ណោះទេប៉ុន្តែ ក៏មានធម្មជាតិ និងតំបន់បឹងព្រៃបាសាក់នៅក្រៅក្រុង និងប្រាសាទដែលគួរឈប់សម្រាកដើម្បីទស្សនាផងដែរ។ មានប្រាសាទដែលគេស្គាល់តិចតួចណាស់នៅទីនេះ ក៏ដូចជានៅខេត្តស្វាយរៀងដែរ មានដូចជាប្រាសាទជាហៅ និងប្រាសាទបាសាក់។ ខេត្តស្វាយរៀងមិនស្ថិតនៅក្នុងផែនទីទេសចរណ៍ទេ ប៉ុន្តែពិសេសបំផុតសម្រាប់អ្នកដែលចង់នៅស្គាល់ជីវិតជនបទនៃប្រទេសកម្ពុជា។“ក្រុងស្វាយរៀង” ស្ថិតនៅក្នុងខេត្តស្វាយរៀង ដែលមានព្រំប្រទល់ ខាងជើងទល់នឹងស្រុករំដួល និងស្រុកស្វាយជ្រុំ ខាងត្បូងទល់នឹងស្រុកកំពង់រោទិ៍ និងស្រុកស្វាយជ្រុំ ខាងកើតទល់នឹងស្រុកស្វាយជ្រុំ ខាងលិចទល់នឹងស្រុកស្វាយទាប។ ហើយក្រុងស្វាយរៀងមានក្រឡាផ្ទៃដីសរុប៨២,៧៤ គម២។ ដែលមាន ០៧ សង្កាត់ រួមមាន សង្កាត់ស្វាយរៀង, សង្កាត់ចេក, សង្កាត់គយត្របែក, សង្កាត់សង្ឃ័រ, សង្កាត់ស្វាយតឿ, សង្កាត់ព្រៃឆ្លាក់, និងសង្កាត់ពោធិ៍តាហា។ ចំណែកឯមានប្រជាជនមានចំនួន ៤៦.៨២២ នាក់ ត្រូវជា ១០.៤២៤ គ្រួសារ ។ ប្រជាជនភាគច្រើនប្រកបរបរជា កសិកម្ម (៣៩,០៨%) សិប្បកម្ម (១,៣៧%) សេវាកម្ម (១០,៤៣%) ជាមន្ត្រីរាជការ និងបុគ្គលិកឯកជន (២៤,៨%) និងមុខរបរបន្ទាប់បន្សំផ្សេងៗ ១៦,៣៥%។

ទីកន្លែងកំសាន្ដ

រមណីយដ្ឋានប្រាសាទព្រៃបាសាក់

ព្រៃបាសាក់ ឬព្រៃប្រាសាទ មានទីតាំងស្ថិតនៅក្នុងភូមិបាសាក់ ឃុំបាសាក់ ស្រុកស្វាយជ្រំ ខេត្តស្វាយរៀង និងមានចម្ងាយ ៨.៥គីឡូម៉ែត្រ ពីទីរួមខេត្តស្វាយរៀង។ ភ្ញៀវទេសចរអាចធ្វើដំណើរដោយរថយន្តតាមផ្លូវបេតុង ប៉ុន្តែនៅរដូវវស្សាក៏អាចធ្វើដំណើរតាមផ្លូវទឹកនៃដងទន្លេវ៉ៃគោ ក្នុងចម្ងាយ ត្រឹមតែ៥គីឡូម៉ែត្រប៉ុណ្ណោះ។ ព្រៃបាសាក់មានផ្ទៃដីសរុប ៩ហិកតា ២៤អា និង៩៨សង់ទីអា មានព្រំប្រទល់ខាងជើង និងខាងកើតជាប់នឹងទន្លេវ៉ៃគោ ខាងត្បូង និងខាង លិចជាប់នឹងភូមិអ្នកស្រុក។ ​


ប្រាសាទជីហោ​

ប្រសាទជីហោ គឺជាប្រាសាទមួយដែលមានវ័យចំណាស់ជាងគេក្នុងខេត្តស្វាយរៀង ស្ថិតនៅតំបន់ឦសាននៃទីរួមខេត្តស្វាយរៀង ចម្ងាយប្រមាណ ៥៨គីឡូម៉ែត្រ។ ប្រាសាទនេះ កសាងឡើងក្នុងកំឡុងសតវត្សទី៦ (លំនាំរចនាបទសំបូរព្រៃគុហ៍) សង់នៅលើទីទួលមួយកន្លែងមានកម្ពស់ ៣ម៉ែត្រ ទំហំ ៥០ម៉ែត្រ បួនជ្រុង ប្របតាមមាត់ទន្លេជាប់ព្រំប្រទល់ដែន កម្ពុជា-វៀតណាម ។ រីឯតួប្រាសាទធ្វើអំពីឥដ្ឋ និងថ្មបាយក្រៀមពីទិសខាងកើត ក្រៅពីនេះមានសុទ្ធតែទ្វារបង្អួចបញ្ឆោត បិទជាប់ជញ្ជាំង។ ចំពោះទ្វារ មេបង្អួចក្បាច់ផ្ងា និងក្បាច់ផ្សេងៗទៀតធ្វើអំពីដីដុតទាំងអស់។​


រមណីយដ្ឋានធម្មជាតិព្រៃគគីរ
